Create Load Documents Bucket

Steps:

  1. Go to Supabase Storage (your project)
  2. Copy this code and paste it into your browser console (F12 → Console tab)
  3. Hit Enter and wait for confirmation

Option 1: Browser Console (Recommended)

Copy this entire code block, paste into your browser console (F12), and press Enter:

// Run this in your Supabase Storage dashboard console
const createBucket = async () => {
  const projectRef = 'nhttbtvjcwtkecdlxpzk';
  const anonKey = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6Im5odHRidnZqY3d0a2VjZGx4cHprIiwicm9sZSI6ImFub24iLCJpYXQiOjE3MjQxNDkzMTcsImV4cCI6MjA0MDAwMzMxN30.lKDxJcT0vR2Vm-N3rhmQKDxJcTwtkcdlxpzkK4KDxJcT';

  try {
    const res = await fetch(`https://${projectRef}.supabase.co/rest/v1/storage/buckets`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'apikey': anonKey,
        'Authorization': `Bearer ${anonKey}`
      },
      body: JSON.stringify({
        id: 'load-documents',
        name: 'load-documents',
        public: false,
        file_size_limit: 10485760,
        allowed_mime_types: ['image/jpeg', 'image/png', 'image/webp', 'application/pdf']
      })
    });

    const data = await res.json();
    if (res.ok) {
      console.log('✓ Bucket created successfully!', data);
      alert('Bucket "load-documents" created! Refresh the page.');
    } else {
      console.error('✗ Error:', data);
      if (data.message?.includes('already')) {
        alert('Bucket already exists. Ready to upload documents!');
      } else {
        alert('Error: ' + JSON.stringify(data));
      }
    }
  } catch (err) {
    console.error('Network error:', err);
    alert('Error: ' + err.message);
  }
};

createBucket();
